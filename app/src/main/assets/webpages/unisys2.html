2. What is the output of the program
void main()
{

		int i,j,k;

		i=2;

		j=4;

		k=i++>j&2;

		printf("%d\n",k);

		if(++k && ++i<--j||i++)

		{

			j=++k;

		}

		printf("%d%d%d",i,-j--,k);

		getch();

	}
// A. 4,-3,2
// B. 5,-3,2
// c. 4,-2,2
// D. 5,-2,2
Ans: D
